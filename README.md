  |     | - _app
  |     | - _document
  |     | - index
  |     | - 404
  |     | - 500
  | - ui (colors, sizes, components)
  |     | - components
  |     |       | - ??

  |     |           | - styles
  |     |           | - ?? (childrens)

  |     | - layouts
